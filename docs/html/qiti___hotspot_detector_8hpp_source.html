29/**
30 Analyzes profiling data to identify performance hotspots.
31
32 Uses call frequency and timing data to automatically detect functions that
33 contribute most significantly to overall execution time.
34 */
35class HotspotDetector
36{
37public:
38 /**
39 Represents a detected performance hotspot.
40 */
41 struct Hotspot
42 {
43 const FunctionData* function = nullptr; ///< The function identified as a hotspot
44 double score = 0.0; ///< Hotspot score (higher = more significant)
45 std::string reason; ///< Human-readable explanation of why this is a hotspot
46
47 // Explicitly define destructor to prevent instrumentation
48 QITI_API ~Hotspot() noexcept = default;
49 };
50
51 /**
52 @returns A vector of detected hotspots, sorted by score (highest first).
53
54 Analyzes all currently profiled functions and identifies those that consume
55 the most execution time based on call frequency and average execution time.
56 Functions with higher total time consumption receive higher scores.
57 */
58 [[nodiscard]] QITI_API static std::vector<Hotspot> detectHotspots() noexcept;
59
60 /**
61 @returns A vector of detected hotspots above the specified threshold.
62
63 @param scoreThreshold Minimum score required for a function to be considered a hotspot.
64 Functions with scores below this threshold will be filtered out.
65 */
66 [[nodiscard]] QITI_API static std::vector<Hotspot> detectHotspots(double scoreThreshold) noexcept;
67
68private:
69 /**
70 Calculate the hotspot score for a given function.
71
72 @param func The function to analyze
73 @returns A hotspot score based on total time consumption (calls Ã— average time)
74 */
75 [[nodiscard]] QITI_API_INTERNAL static double calculateHotspotScore(const FunctionData* func) noexcept;
76
77 /**
78 Generate a human-readable explanation for why a function is considered a hotspot.
79
80 @param func The function to analyze
81 @returns A descriptive string explaining the hotspot characteristics
82 */
83 [[nodiscard]] QITI_API_INTERNAL static std::string getHotspotReason(const FunctionData* func) noexcept;
84};
